def matxMultiply(A, B):
    A_col_num = shape(A)[1]
    B_row_num = shape(B)[0]
    if A_col_num != B_row_num: # AB
        raise ValueError
    product = list()
    A_row_num = shape(A)[0]
    B_col_num = shape(B)[1]
    for i in range(A_row_num):
        row = []
        for j in range(B_col_num):
            tmp = sum([A[i][k] * B[k][j] for k in range(A_col_num)])
            row.append(tmp)
        product.append(row)
    return product

""" Gaussian Jordan  Ax = b.
    
        A:

        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
    return abs(val) < epsilon
"""
def gj_Solve(A, b, decPts=4, epsilon = 1.0e-16):
    #1 Ab
    if shape(A)[0] != shape(b)[0]:
        raise ValueError
    #2 Ab
    Ab = augmentMatrix(A, b)
    # 3 Ab
    # Abx
    for i in range(shape(A)[0]):
        # 
        tmp_max_line = i
        for l in range(i, shape(A)[0]): 
            if abs(Ab[l][i]) > abs(Ab[tmp_max_line][i]):
                tmp_max_line = l
        if abs(Ab[tmp_max_line][i]) < epsilon:
            return None
        swapRows(Ab, i, tmp_max_line) 
        if Ab[i][i] != 1:
            if abs(Ab[i][i]) > epsilon:
                scaleRow(Ab, i, 1/Ab[i][i]) # 
            else:
                continue
        for j in range(shape(A)[0]):
            # Abi+1y
            if i!= j:
                addScaledRow(Ab, j, i, -Ab[j][i])
    matxRound(Ab, decPts)
    result = []
    for col in range(shape(A)[0]):
        result.append([Ab[col][-1]])
    return result
