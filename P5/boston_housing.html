def performance_metric2(y_true, y_predict):
    """"""
    y_mean = sum(y_true)/len(y_true)
    sst = sum(map(lambda x:(x-y_mean)**2, y_true))
    ssr = sum([(x-y)**2 for x, y in zip(y_true, y_predict)])
    score = 1- ssr/sst
    return score

    params = {'max_depth':list(range(1,11))}
