def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    if city == 'Washington':
        starttime_str = datum['Start date']
    else:
        starttime_str = datum['starttime']
    
    # 
    if city == 'NYC':
        starttime = datetime.strptime(starttime_str,'%m/%d/%Y %H:%M:%S')
    else:
        starttime = datetime.strptime(starttime_str,'%m/%d/%Y %H:%M')
    month = starttime.month
    hour = starttime.hour
    day_of_week = datetime.strftime(starttime, '%A')

    return (month, hour, day_of_week)

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)

        # 
        for row in trip_reader:
            # 
            new_point = {}

            ## ##
            duration = duration_in_mins(row, city)
            month, hour, day_of_week = time_of_trip(row, city)
            user_type = type_of_user(row, city)
            
            
            ##  new_point  ##
            ##  DictWriter         ##
            new_point['duration'] = duration
            new_point['month'] = month
            new_point['hour'] = hour
            new_point['day_of_week'] = day_of_week
            new_point['user_type'] = user_type

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def time_of_bike(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_total = 0
        n_total_time = 0
        n_over_half_hour = 0
        
        # 
        for row in reader:
            duration = float(row['duration'])
            n_total = n_total + 1
            n_total_time = n_total_time + duration 
            if duration > 30.0:
                n_over_half_hour += 1

        
        # 
        n_avg_bike = n_total_time / n_total
        #  
        n_over_half_hour_scale = n_over_half_hour / n_total
        
        
        # 
        return(n_avg_bike, n_over_half_hour_scale)

def list_sort(freq_band):
    freq_band_sorted = sorted(freq_band.items(),key = lambda x:int(float(x[0])))
    return freq_band_sorted

def freq_band_value(freq_band_sort):
    l_freq_band_value = []
    for k,v in freq_band_sort:
        l_freq_band_value.append(v)
    return l_freq_band_value


def freq_band_time(city):
    data_file = city_info[city]['out_file']
    with open(data_file, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        tmp_freq_band = {}
        tmp_freq_band_customers = {}
        tmp_freq_band_subscriber = {}
        # 
        for row in reader:
            hour = row['hour']
            if hour in tmp_freq_band:
                tmp_freq_band[hour] += 1
            else:
                tmp_freq_band[hour] = 1
            if row['user_type'] == 'Subscriber':
                if hour in tmp_freq_band_subscriber:
                    tmp_freq_band_subscriber[hour] += 1
                else:
                    tmp_freq_band_subscriber[hour] = 1
            else:
                    if hour in tmp_freq_band_customers:
                        tmp_freq_band_customers[hour] += 1
                    else:
                        tmp_freq_band_customers[hour] = 1
        l1 = list_sort(tmp_freq_band)
        l_freq_band = freq_band_value(l1)
        l2 = list_sort(tmp_freq_band_customers)
        l_freq_band_customers = freq_band_value(l2)
        l3 =  list_sort(tmp_freq_band_subscriber)
        l_freq_band_subscriber = freq_band_value(l3)
        return (l_freq_band, l_freq_band_customers, l_freq_band_subscriber)
